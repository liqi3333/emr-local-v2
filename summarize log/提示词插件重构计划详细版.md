  if (!text) return '';
  const disease = context.disease || '';
  const patientInfo = context.patientInfo || {};

  const replacements = {
    '{{disease}}': disease,
    '{{patientContext}}': buildContextString('患者基本信息', patientInfo),

    '{{attendingDataContext}}': buildContextString('主治医师查房记录', context.attendingData),

    '{{dischargeDataContext}}': buildContextString('出院小结内容', context.dischargeData),
    '{{surgeryConsentDataContext}}': buildContextString('手术同意书内容', context.surgeryConsentData),
    '{{bloodTransfusionConsentDataContext}}': buildContextString('输血同意书内容', context.bloodTransfusionConsentData),
    '{{anesthesiaConsentDataContext}}': buildContextString('麻醉同意书内容', context.anesthesiaConsentData),

    '{{nursingRecordSheetDataContext}}': buildContextString('护理记录单内容', context.nursingRecordSheetData),
  };

  let result = text;

### 4.1 完善 Diff 检测（30-45 分钟）

**文件：** `public/js/components/PromptEditor.js`

**操作：** 重写 `_showDiff()`。

```javascript
_showDiff() {
  const current = this._state.currentTemplate;
  const defaultTmpl = this._state.defaultTemplate;
  if (!current || !defaultTmpl) return;

  const registry = this._state.registry;
  const diffLines = [];

  for (const cat of registry.categories) {
    for (const type of cat.types) {
      const typeKey = type.templateKey;
      const defaultType = defaultTmpl.templates?.[typeKey] || {};
      const currentType = current.templates?.[typeKey] || {};

      // 检测类型级提示词变化
      const textKeys = ['rolePrompt', 'outputFormat', 'endingPrompt', 'userPrompt'];
      for (const key of textKeys) {
        const d = defaultType[key] || '';
        const c = currentType[key] || '';
        if (d !== c) {
          diffLines.push(`[${key} 修改] ${typeKey}`);
        }
      }

      // 检测字段变化
      const defaultFields = defaultType.fields || {};
      const currentFields = currentType.fields || {};
      const registryFields = type.fields || [];

      // 字段删除
      for (const key of Object.keys(currentFields)) {
        if (!registryFields.find((f) => f.key === key)) {
          diffLines.push(`[字段删除] ${typeKey}.${key}`);
        }
      }

      // 字段新增/修改
      for (const field of registryFields) {
        if (field.enabled === false) continue;
        const d = defaultFields[field.key];
        const c = currentFields[field.key];
        if (!c && d) {
          diffLines.push(`[新增字段覆盖] ${typeKey}.${field.key}`);
        } else if (c) {
          if (c.label !== field.label) {
            diffLines.push(`[字段 label 修改] ${typeKey}.${field.key}`);
          }
          if (c.description && c.description !== field.description) {
            diffLines.push(`[字段 description 修改] ${typeKey}.${field.key}`);
          }
        }
      }

      // 新增类型
      if (!current.templates?.[typeKey]) {
        diffLines.push(`[新增类型] ${typeKey}: ${type.label}`);
      }
    }
  }

  alert(diffLines.length ? diffLines.join('\n') : '当前模板与默认模板一致');
}
```

**验证：** 修改 rolePrompt、字段 description、删除字段后，Diff 都能正确显示。

### 7.2 syncTemplate 感知 Registry 字段变化（30-45 分钟）

**文件：** `src/services/promptTemplates.js`

**操作：** 修改 `syncTemplate(name)`：

```javascript
function syncTemplate(name) {
  if (name === 'default') return { template: getDefaultTemplate(), changed: false };

  const defaultTemplate = getDefaultTemplate();
  const customTemplate = getTemplate(name);
  const registry = require('./recordRegistry').getRegistry() || require('../data/recordRegistry').DEFAULT_REGISTRY;
  const originalVersion = customTemplate.defaultVersion || '';
  let changed = false;

  for (const [typeKey, defaultType] of Object.entries(defaultTemplate.templates)) {
    if (!customTemplate.templates[typeKey]) {
      customTemplate.templates[typeKey] = deepClone(defaultType);
      changed = true;
      continue;
    }

    const customType = customTemplate.templates[typeKey];
    const registryType = findRegistryType(registry, typeKey);

    // 同步字段：以 Registry 为基准
    const syncedFields = {};
    for (const field of registryType?.fields || []) {
      if (field.enabled === false) continue;
      if (!customType.fields?.[field.key]) {
        syncedFields[field.key] = { label: field.label, description: '' };
        changed = true;
      } else {
        syncedFields[field.key] = customType.fields[field.key];
      }
    }
    customType.fields = syncedFields;
  }

  const newVersion = getCompositeVersion();
  if (customTemplate.defaultVersion !== newVersion) {
    customTemplate.defaultVersion = newVersion;
    changed = true;
  }

  if (changed) saveTemplate(name, customTemplate);
  return { template: customTemplate, changed, previousVersion: originalVersion, currentVersion: newVersion };
}
```

**验证：** Registry 删除字段后 sync，自定义模板中对应字段覆盖也被删除。
